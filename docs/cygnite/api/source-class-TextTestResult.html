315:   function TestSuite($classname=false) {
316:     // Find all methods of the given class whose name starts with
317:     // "test" and add them to the test suite.
318: 
319:     // PHP3: We are just _barely_ able to do this with PHP's limited
320:     // introspection...  Note that PHP seems to store method names in
321:     // lower case, and we have to avoid the constructor function for
322:     // the TestCase class superclass.  Names of subclasses of TestCase
323:     // must not start with "Test" since such a class will have a
324:     // constructor method name also starting with "test" and we can't
325:     // distinquish such a construtor from the real test method names.
326:     // So don't name any TestCase subclasses as "Test..."!
327: 
328:     // PHP4:  Never mind all that.  We can now ignore constructor
329:     // methods, so a test class may be named "Test...".
330: 
331:     if (empty($classname))
332:       return;
333:     $this->fClassname = $classname;
334: 
335:     if (floor(phpversion()) >= 4) {
336:       // PHP4 introspection, submitted by Dylan Kuhn
337: 
338:       $names = get_class_methods($classname);
339:       while (list($key, $method) = each($names)) {
340:         if (preg_match('/^test/', $method)) {
341:           $test = new $classname($method);
342:           if (strcasecmp($method, $classname) == 0 || is_subclass_of($test, $method)) {
343:             // Ignore the given method name since it is a constructor:
344:             // it's the name of our test class or it is the name of a
345:             // superclass of our test class.  (This code smells funny.
346:             // Anyone got a better way?)
347: 
348:             //print "skipping $method<br>";
349:           }
350:           else {
351:             $this->addTest($test);
352:           }
353:         }
354:       }
355:     }
356:     else {  // PHP3
357:       $dummy = new $classname("dummy");
358:       $names = (array) $dummy;
359:       while (list($key, $value) = each($names)) {
360:         $type = gettype($value);
361:         if ($type == "user function" && preg_match('/^test/', $key)
